# 添加模块

Entity Engine 采用模块化架构，支持通过插件系统动态加载和扩展功能模块。开发者可以创建自定义模块来添加新功能、集成第三方服务或实现特定业务逻辑。

## 模块架构

### 模块接口

```typescript
/**
 * Entity Engine模块接口
 */
interface IEntityModule {
  // 基础信息
  name: string;
  version: string;
  displayName: string;
  description?: string;
  author?: string;
  
  // 依赖关系
  dependencies?: ModuleDependency[];
  optionalDependencies?: ModuleDependency[];
  peerDependencies?: ModuleDependency[];
  
  // 生命周期
  initialize(context: ModuleContext): Promise<void> | void;
  activate(context: ModuleContext): Promise<void> | void;
  deactivate(context: ModuleContext): Promise<void> | void;
  dispose(context: ModuleContext): Promise<void> | void;
  
  // 功能提供
  provides?: ModuleProvides;
  
  // 配置
  getDefaultConfig(): ModuleConfig;
  validateConfig(config: ModuleConfig): ConfigValidationResult;
  
  // 健康检查
  healthCheck?(): Promise<HealthStatus>;
}

/**
 * 模块依赖定义
 */
interface ModuleDependency {
  name: string;
  version?: string;
  optional?: boolean;
}

/**
 * 模块提供的功能
 */
interface ModuleProvides {
  fieldTypes?: string[];
  viewTypes?: string[];
  actions?: string[];
  events?: string[];
  services?: string[];
  middleware?: string[];
  renderers?: string[];
}

/**
 * 模块上下文
 */
interface ModuleContext {
  // 核心服务
  entityManager: IEntityManager;
  fieldTypeRegistry: IFieldTypeRegistry;
  viewTypeRegistry: IViewTypeRegistry;
  actionManager: IActionManager;
  eventManager: IEventManager;
  
  // 模块管理
  moduleManager: IModuleManager;
  moduleConfig: ModuleConfig;
  
  // 应用上下文
  applicationContext: ApplicationContext;
  logger: ILogger;
}

/**
 * 模块配置
 */
interface ModuleConfig {
  enabled: boolean;
  autoStart: boolean;
  priority: number;
  settings: Record<string, any>;
}
```

### 模块管理器

```typescript
/**
 * 模块管理器接口
 */
interface IModuleManager {
  // 模块注册
  register(module: IEntityModule): Promise<void>;
  unregister(moduleName: string): Promise<void>;
  
  // 模块控制
  activate(moduleName: string): Promise<void>;
  deactivate(moduleName: string): Promise<void>;
  
  // 模块查询
  list(): IEntityModule[];
  get(moduleName: string): IEntityModule | undefined;
  isActive(moduleName: string): boolean;
  
  // 依赖管理
  resolveDependencies(moduleName: string): string[];
  checkDependencies(module: IEntityModule): DependencyCheckResult;
  
  // 生命周期
  initializeAll(): Promise<void>;
  activateAll(): Promise<void>;
  deactivateAll(): Promise<void>;
  shutdown(): Promise<void>;
}

/**
 * 默认模块管理器实现
 */
class DefaultModuleManager implements IModuleManager {
  private modules = new Map<string, ModuleInfo>();
  private loadOrder: string[] = [];
  private context: ApplicationContext;
  
  constructor(context: ApplicationContext) {
    this.context = context;
  }
  
  async register(module: IEntityModule): Promise<void> {
    // 验证模块
    this.validateModule(module);
    
    // 检查依赖
    const dependencyCheck = this.checkDependencies(module);
    if (!dependencyCheck.satisfied) {
      throw new ModuleError(
        `Module '${module.name}' dependencies not satisfied: ${dependencyCheck.missing.join(', ')}`
      );
    }
    
    // 验证配置
    const defaultConfig = module.getDefaultConfig();
    const configValidation = module.validateConfig(defaultConfig);
    if (!configValidation.valid) {
      throw new ModuleError(
        `Module '${module.name}' default config invalid: ${configValidation.errors.join(', ')}`
      );
    }
    
    // 注册模块
    const moduleInfo: ModuleInfo = {
      module,
      config: { ...defaultConfig, ...this.getStoredConfig(module.name) },
      state: ModuleState.REGISTERED,
      registeredAt: new Date(),
      activatedAt: null,
      error: null
    };
    
    this.modules.set(module.name, moduleInfo);
    
    // 重新计算加载顺序
    this.recalculateLoadOrder();
    
    // 如果配置为自动启动，则初始化和激活
    if (moduleInfo.config.autoStart) {
      await this.initialize(module.name);
      await this.activate(module.name);
    }
    
    // 触发事件
    this.emitEvent('moduleRegistered', { module });
  }
  
  async unregister(moduleName: string): Promise<void> {
    const moduleInfo = this.modules.get(moduleName);
    if (!moduleInfo) {
      throw new ModuleError(`Module '${moduleName}' not found`);
    }
    
    // 检查反向依赖
    const dependents = this.findDependents(moduleName);
    if (dependents.length > 0) {
      throw new ModuleError(
        `Cannot unregister module '${moduleName}': required by ${dependents.join(', ')}`
      );
    }
    
    // 停用和销毁模块
    if (moduleInfo.state === ModuleState.ACTIVE) {
      await this.deactivate(moduleName);
    }
    
    if (moduleInfo.state === ModuleState.INITIALIZED) {
      await this.dispose(moduleName);
    }
    
    // 移除注册
    this.modules.delete(moduleName);
    this.loadOrder = this.loadOrder.filter(name => name !== moduleName);
    
    // 触发事件
    this.emitEvent('moduleUnregistered', { moduleName });
  }
  
  async activate(moduleName: string): Promise<void> {
    const moduleInfo = this.modules.get(moduleName);
    if (!moduleInfo) {
      throw new ModuleError(`Module '${moduleName}' not found`);
    }
    
    if (moduleInfo.state === ModuleState.ACTIVE) {
      return; // 已经激活
    }
    
    try {
      // 确保依赖已激活
      await this.activateDependencies(moduleInfo.module);
      
      // 初始化模块（如果未初始化）
      if (moduleInfo.state === ModuleState.REGISTERED) {
        await this.initialize(moduleName);
      }
      
      // 创建模块上下文
      const moduleContext = this.createModuleContext(moduleInfo);
      
      // 激活模块
      await moduleInfo.module.activate(moduleContext);
      
      // 更新状态
      moduleInfo.state = ModuleState.ACTIVE;
      moduleInfo.activatedAt = new Date();
      moduleInfo.error = null;
      
      // 触发事件
      this.emitEvent('moduleActivated', { module: moduleInfo.module });
      
    } catch (error) {
      moduleInfo.error = error;
      moduleInfo.state = ModuleState.ERROR;
      throw new ModuleError(`Failed to activate module '${moduleName}': ${error.message}`);
    }
  }
  
  async deactivate(moduleName: string): Promise<void> {
    const moduleInfo = this.modules.get(moduleName);
    if (!moduleInfo) {
      throw new ModuleError(`Module '${moduleName}' not found`);
    }
    
    if (moduleInfo.state !== ModuleState.ACTIVE) {
      return; // 未激活
    }
    
    try {
      // 检查反向依赖
      const activeDependents = this.findDependents(moduleName).filter(name => 
        this.isActive(name)
      );
      
      if (activeDependents.length > 0) {
        throw new ModuleError(
          `Cannot deactivate module '${moduleName}': required by active modules ${activeDependents.join(', ')}`
        );
      }
      
      // 创建模块上下文
      const moduleContext = this.createModuleContext(moduleInfo);
      
      // 停用模块
      await moduleInfo.module.deactivate(moduleContext);
      
      // 更新状态
      moduleInfo.state = ModuleState.INITIALIZED;
      moduleInfo.activatedAt = null;
      moduleInfo.error = null;
      
      // 触发事件
      this.emitEvent('moduleDeactivated', { module: moduleInfo.module });
      
    } catch (error) {
      moduleInfo.error = error;
      throw new ModuleError(`Failed to deactivate module '${moduleName}': ${error.message}`);
    }
  }
  
  list(): IEntityModule[] {
    return Array.from(this.modules.values()).map(info => info.module);
  }
  
  get(moduleName: string): IEntityModule | undefined {
    return this.modules.get(moduleName)?.module;
  }
  
  isActive(moduleName: string): boolean {
    return this.modules.get(moduleName)?.state === ModuleState.ACTIVE;
  }
  
  resolveDependencies(moduleName: string): string[] {
    const moduleInfo = this.modules.get(moduleName);
    if (!moduleInfo) {
      throw new ModuleError(`Module '${moduleName}' not found`);
    }
    
    const dependencies = new Set<string>();
    const visited = new Set<string>();
    
    const resolveDeps = (name: string) => {
      if (visited.has(name)) {
        throw new ModuleError(`Circular dependency detected: ${name}`);
      }
      
      visited.add(name);
      
      const info = this.modules.get(name);
      if (!info) return;
      
      const deps = info.module.dependencies || [];
      for (const dep of deps) {
        dependencies.add(dep.name);
        resolveDeps(dep.name);
      }
      
      visited.delete(name);
    };
    
    resolveDeps(moduleName);
    
    return Array.from(dependencies);
  }
  
  checkDependencies(module: IEntityModule): DependencyCheckResult {
    const result: DependencyCheckResult = {
      satisfied: true,
      missing: [],
      optional: []
    };
    
    const dependencies = module.dependencies || [];
    
    for (const dep of dependencies) {
      const depModule = this.modules.get(dep.name);
      if (!depModule) {
        if (dep.optional) {
          result.optional.push(dep.name);
        } else {
          result.satisfied = false;
          result.missing.push(dep.name);
        }
      } else if (dep.version) {
        // 检查版本兼容性
        if (!this.isVersionCompatible(depModule.module.version, dep.version)) {
          result.satisfied = false;
          result.missing.push(`${dep.name}@${dep.version}`);
        }
      }
    }
    
    return result;
  }
  
  async initializeAll(): Promise<void> {
    const errors: Error[] = [];
    
    for (const moduleName of this.loadOrder) {
      try {
        const moduleInfo = this.modules.get(moduleName);
        if (moduleInfo && moduleInfo.state === ModuleState.REGISTERED) {
          await this.initialize(moduleName);
        }
      } catch (error) {
        errors.push(error);
      }
    }
    
    if (errors.length > 0) {
      throw new ModuleError(`Failed to initialize some modules: ${errors.map(e => e.message).join('; ')}`);
    }
  }
  
  async activateAll(): Promise<void> {
    const errors: Error[] = [];
    
    for (const moduleName of this.loadOrder) {
      try {
        const moduleInfo = this.modules.get(moduleName);
        if (moduleInfo && moduleInfo.config.autoStart && moduleInfo.state !== ModuleState.ACTIVE) {
          await this.activate(moduleName);
        }
      } catch (error) {
        errors.push(error);
      }
    }
    
    if (errors.length > 0) {
      throw new ModuleError(`Failed to activate some modules: ${errors.map(e => e.message).join('; ')}`);
    }
  }
  
  async deactivateAll(): Promise<void> {
    // 按逆序停用
    const reverseOrder = [...this.loadOrder].reverse();
    
    for (const moduleName of reverseOrder) {
      try {
        const moduleInfo = this.modules.get(moduleName);
        if (moduleInfo && moduleInfo.state === ModuleState.ACTIVE) {
          await this.deactivate(moduleName);
        }
      } catch (error) {
        console.error(`Failed to deactivate module '${moduleName}':`, error);
      }
    }
  }
  
  async shutdown(): Promise<void> {
    await this.deactivateAll();
    
    // 销毁所有模块
    for (const moduleName of [...this.loadOrder].reverse()) {
      try {
        await this.dispose(moduleName);
      } catch (error) {
        console.error(`Failed to dispose module '${moduleName}':`, error);
      }
    }
    
    this.modules.clear();
    this.loadOrder = [];
  }
  
  // 私有方法
  private async initialize(moduleName: string): Promise<void> {
    const moduleInfo = this.modules.get(moduleName);
    if (!moduleInfo) return;
    
    try {
      const moduleContext = this.createModuleContext(moduleInfo);
      await moduleInfo.module.initialize(moduleContext);
      moduleInfo.state = ModuleState.INITIALIZED;
    } catch (error) {
      moduleInfo.error = error;
      moduleInfo.state = ModuleState.ERROR;
      throw error;
    }
  }
  
  private async dispose(moduleName: string): Promise<void> {
    const moduleInfo = this.modules.get(moduleName);
    if (!moduleInfo) return;
    
    try {
      const moduleContext = this.createModuleContext(moduleInfo);
      await moduleInfo.module.dispose(moduleContext);
      moduleInfo.state = ModuleState.REGISTERED;
    } catch (error) {
      moduleInfo.error = error;
      throw error;
    }
  }
  
  private validateModule(module: IEntityModule): void {
    if (!module.name || typeof module.name !== 'string') {
      throw new ModuleError('Module must have a valid name');
    }
    
    if (!module.version || typeof module.version !== 'string') {
      throw new ModuleError('Module must have a valid version');
    }
    
    if (this.modules.has(module.name)) {
      throw new ModuleError(`Module '${module.name}' already registered`);
    }
    
    if (typeof module.initialize !== 'function') {
      throw new ModuleError('Module must implement initialize method');
    }
    
    if (typeof module.activate !== 'function') {
      throw new ModuleError('Module must implement activate method');
    }
    
    if (typeof module.deactivate !== 'function') {
      throw new ModuleError('Module must implement deactivate method');
    }
    
    if (typeof module.dispose !== 'function') {
      throw new ModuleError('Module must implement dispose method');
    }
  }
  
  private recalculateLoadOrder(): void {
    const modules = Array.from(this.modules.keys());
    const loadOrder: string[] = [];
    const visited = new Set<string>();
    const visiting = new Set<string>();
    
    const visit = (moduleName: string) => {
      if (visiting.has(moduleName)) {
        throw new ModuleError(`Circular dependency detected: ${moduleName}`);
      }
      
      if (visited.has(moduleName)) {
        return;
      }
      
      visiting.add(moduleName);
      
      const moduleInfo = this.modules.get(moduleName);
      if (moduleInfo) {
        const deps = moduleInfo.module.dependencies || [];
        for (const dep of deps) {
          if (this.modules.has(dep.name)) {
            visit(dep.name);
          }
        }
      }
      
      visiting.delete(moduleName);
      visited.add(moduleName);
      loadOrder.push(moduleName);
    };
    
    for (const moduleName of modules) {
      visit(moduleName);
    }
    
    this.loadOrder = loadOrder;
  }
  
  private async activateDependencies(module: IEntityModule): Promise<void> {
    const deps = module.dependencies || [];
    
    for (const dep of deps) {
      if (!dep.optional && !this.isActive(dep.name)) {
        await this.activate(dep.name);
      }
    }
  }
  
  private findDependents(moduleName: string): string[] {
    const dependents: string[] = [];
    
    for (const [name, moduleInfo] of this.modules) {
      const deps = moduleInfo.module.dependencies || [];
      if (deps.some(dep => dep.name === moduleName && !dep.optional)) {
        dependents.push(name);
      }
    }
    
    return dependents;
  }
  
  private createModuleContext(moduleInfo: ModuleInfo): ModuleContext {
    return {
      entityManager: this.context.entityManager,
      fieldTypeRegistry: this.context.fieldTypeRegistry,
      viewTypeRegistry: this.context.viewTypeRegistry,
      actionManager: this.context.actionManager,
      eventManager: this.context.eventManager,
      moduleManager: this,
      moduleConfig: moduleInfo.config,
      applicationContext: this.context,
      logger: this.context.loggerFactory.createLogger(moduleInfo.module.name)
    };
  }
  
  private getStoredConfig(moduleName: string): Partial<ModuleConfig> {
    // 从配置存储加载模块配置
    return {};
  }
  
  private isVersionCompatible(actual: string, required: string): boolean {
    // 实现版本兼容性检查逻辑
    return true;
  }
  
  private emitEvent(eventName: string, data: any): void {
    // 触发事件
    this.context.eventManager.emit(`module.${eventName}`, data);
  }
}
```

## 自定义模块示例

### 审计模块

```typescript
/**
 * 审计模块 - 记录实体操作历史
 */
class AuditModule implements IEntityModule {
  name = 'audit';
  version = '1.0.0';
  displayName = '审计模块';
  description = '记录和追踪实体的创建、更新和删除操作';
  author = 'Entity Engine Team';
  
  dependencies = [];
  
  private auditLogger?: IAuditLogger;
  private eventListeners: Array<() => void> = [];
  
  async initialize(context: ModuleContext): Promise<void> {
    // 初始化审计日志记录器
    this.auditLogger = new DatabaseAuditLogger({
      tableName: 'entity_audit_log',
      connectionString: context.moduleConfig.settings.databaseUrl
    });
    
    await this.auditLogger.initialize();
    
    context.logger.info('Audit module initialized');
  }
  
  async activate(context: ModuleContext): Promise<void> {
    if (!this.auditLogger) {
      throw new Error('Audit logger not initialized');
    }
    
    // 监听实体事件
    const eventManager = context.eventManager;
    
    // 监听创建事件
    const onEntityCreated = async (event: EntityCreatedEvent) => {
      await this.auditLogger!.logCreate({
        entityType: event.entityType,
        entityId: event.entityId,
        data: event.data,
        timestamp: event.timestamp,
        userId: event.userId,
        metadata: event.metadata
      });
    };
    
    // 监听更新事件
    const onEntityUpdated = async (event: EntityUpdatedEvent) => {
      await this.auditLogger!.logUpdate({
        entityType: event.entityType,
        entityId: event.entityId,
        changes: event.changes,
        previousData: event.previousData,
        newData: event.newData,
        timestamp: event.timestamp,
        userId: event.userId,
        metadata: event.metadata
      });
    };
    
    // 监听删除事件
    const onEntityDeleted = async (event: EntityDeletedEvent) => {
      await this.auditLogger!.logDelete({
        entityType: event.entityType,
        entityId: event.entityId,
        data: event.data,
        timestamp: event.timestamp,
        userId: event.userId,
        metadata: event.metadata
      });
    };
    
    // 注册事件监听器
    eventManager.on('entity.created', onEntityCreated);
    eventManager.on('entity.updated', onEntityUpdated);
    eventManager.on('entity.deleted', onEntityDeleted);
    
    // 保存清理函数
    this.eventListeners.push(
      () => eventManager.off('entity.created', onEntityCreated),
      () => eventManager.off('entity.updated', onEntityUpdated),
      () => eventManager.off('entity.deleted', onEntityDeleted)
    );
    
    // 注册审计查询API
    this.registerAuditAPI(context);
    
    context.logger.info('Audit module activated');
  }
  
  async deactivate(context: ModuleContext): Promise<void> {
    // 清理事件监听器
    this.eventListeners.forEach(cleanup => cleanup());
    this.eventListeners = [];
    
    context.logger.info('Audit module deactivated');
  }
  
  async dispose(context: ModuleContext): Promise<void> {
    if (this.auditLogger) {
      await this.auditLogger.dispose();
      this.auditLogger = undefined;
    }
    
    context.logger.info('Audit module disposed');
  }
  
  getDefaultConfig(): ModuleConfig {
    return {
      enabled: true,
      autoStart: true,
      priority: 100,
      settings: {
        databaseUrl: process.env.AUDIT_DATABASE_URL || process.env.DATABASE_URL,
        retentionDays: 365,
        includeMetadata: true,
        logLevel: 'info'
      }
    };
  }
  
  validateConfig(config: ModuleConfig): ConfigValidationResult {
    const errors: string[] = [];
    
    if (!config.settings.databaseUrl) {
      errors.push('Database URL is required for audit module');
    }
    
    if (typeof config.settings.retentionDays !== 'number' || config.settings.retentionDays < 1) {
      errors.push('Retention days must be a positive number');
    }
    
    return {
      valid: errors.length === 0,
      errors,
      warnings: []
    };
  }
  
  async healthCheck(): Promise<HealthStatus> {
    if (!this.auditLogger) {
      return {
        status: 'unhealthy',
        message: 'Audit logger not initialized',
        timestamp: new Date()
      };
    }
    
    try {
      await this.auditLogger.healthCheck();
      return {
        status: 'healthy',
        message: 'Audit module is operational',
        timestamp: new Date()
      };
    } catch (error) {
      return {
        status: 'unhealthy',
        message: `Audit logger health check failed: ${error.message}`,
        timestamp: new Date()
      };
    }
  }
  
  private registerAuditAPI(context: ModuleContext): void {
    // 注册审计查询动作
    const auditQueryAction = {
      name: 'audit.query',
      displayName: '查询审计日志',
      description: '查询实体操作审计日志',
      handler: async (params: AuditQueryParams) => {
        if (!this.auditLogger) {
          throw new Error('Audit logger not available');
        }
        
        return await this.auditLogger.query(params);
      },
      inputSchema: {
        type: 'object',
        properties: {
          entityType: { type: 'string' },
          entityId: { type: 'string' },
          operation: { type: 'string', enum: ['create', 'update', 'delete'] },
          userId: { type: 'string' },
          fromDate: { type: 'string', format: 'date-time' },
          toDate: { type: 'string', format: 'date-time' },
          limit: { type: 'number', minimum: 1, maximum: 1000, default: 50 },
          offset: { type: 'number', minimum: 0, default: 0 }
        }
      }
    };
    
    context.actionManager.register(auditQueryAction);
  }
}
```

### 缓存模块

```typescript
/**
 * 缓存模块 - 提供实体数据缓存
 */
class CacheModule implements IEntityModule {
  name = 'cache';
  version = '1.0.0';
  displayName = '缓存模块';
  description = '提供实体数据的内存和持久化缓存';
  
  dependencies = [];
  
  private cacheProvider?: ICacheProvider;
  private eventListeners: Array<() => void> = [];
  
  async initialize(context: ModuleContext): Promise<void> {
    const settings = context.moduleConfig.settings;
    
    // 根据配置选择缓存提供者
    switch (settings.provider) {
      case 'redis':
        this.cacheProvider = new RedisCacheProvider(settings.redis);
        break;
      case 'memory':
        this.cacheProvider = new MemoryCacheProvider(settings.memory);
        break;
      case 'hybrid':
        this.cacheProvider = new HybridCacheProvider({
          l1: new MemoryCacheProvider(settings.memory),
          l2: new RedisCacheProvider(settings.redis)
        });
        break;
      default:
        throw new Error(`Unknown cache provider: ${settings.provider}`);
    }
    
    await this.cacheProvider.initialize();
    
    context.logger.info(`Cache module initialized with ${settings.provider} provider`);
  }
  
  async activate(context: ModuleContext): Promise<void> {
    if (!this.cacheProvider) {
      throw new Error('Cache provider not initialized');
    }
    
    // 注册缓存中间件
    this.registerCacheMiddleware(context);
    
    // 监听实体变化事件以失效缓存
    this.registerCacheInvalidation(context);
    
    // 注册缓存管理动作
    this.registerCacheActions(context);
    
    context.logger.info('Cache module activated');
  }
  
  async deactivate(context: ModuleContext): Promise<void> {
    this.eventListeners.forEach(cleanup => cleanup());
    this.eventListeners = [];
    
    context.logger.info('Cache module deactivated');
  }
  
  async dispose(context: ModuleContext): Promise<void> {
    if (this.cacheProvider) {
      await this.cacheProvider.dispose();
      this.cacheProvider = undefined;
    }
    
    context.logger.info('Cache module disposed');
  }
  
  getDefaultConfig(): ModuleConfig {
    return {
      enabled: true,
      autoStart: true,
      priority: 10,
      settings: {
        provider: 'memory',
        ttl: 3600, // 1 hour
        maxSize: 1000,
        memory: {
          maxSize: 1000,
          ttl: 3600
        },
        redis: {
          host: 'localhost',
          port: 6379,
          db: 0,
          keyPrefix: 'ee:cache:'
        }
      }
    };
  }
  
  validateConfig(config: ModuleConfig): ConfigValidationResult {
    const errors: string[] = [];
    
    const validProviders = ['memory', 'redis', 'hybrid'];
    if (!validProviders.includes(config.settings.provider)) {
      errors.push(`Invalid cache provider. Must be one of: ${validProviders.join(', ')}`);
    }
    
    if (typeof config.settings.ttl !== 'number' || config.settings.ttl < 0) {
      errors.push('TTL must be a non-negative number');
    }
    
    return {
      valid: errors.length === 0,
      errors,
      warnings: []
    };
  }
  
  async healthCheck(): Promise<HealthStatus> {
    if (!this.cacheProvider) {
      return {
        status: 'unhealthy',
        message: 'Cache provider not initialized',
        timestamp: new Date()
      };
    }
    
    try {
      await this.cacheProvider.healthCheck();
      return {
        status: 'healthy',
        message: 'Cache module is operational',
        timestamp: new Date(),
        details: {
          provider: this.cacheProvider.getName(),
          stats: await this.cacheProvider.getStats()
        }
      };
    } catch (error) {
      return {
        status: 'unhealthy',
        message: `Cache provider health check failed: ${error.message}`,
        timestamp: new Date()
      };
    }
  }
  
  private registerCacheMiddleware(context: ModuleContext): void {
    // 查询缓存中间件
    const queryMiddleware = {
      name: 'cache.query',
      before: async (ctx: QueryContext) => {
        if (!ctx.options.useCache) return;
        
        const cacheKey = this.generateCacheKey('query', ctx.entityType, ctx.query);
        const cached = await this.cacheProvider!.get(cacheKey);
        
        if (cached) {
          ctx.result = cached;
          ctx.skipExecution = true;
          ctx.fromCache = true;
        }
      },
      after: async (ctx: QueryContext) => {
        if (ctx.fromCache || !ctx.options.useCache) return;
        
        const cacheKey = this.generateCacheKey('query', ctx.entityType, ctx.query);
        const ttl = ctx.options.cacheTTL || context.moduleConfig.settings.ttl;
        
        await this.cacheProvider!.set(cacheKey, ctx.result, ttl);
      }
    };
    
    context.entityManager.addQueryMiddleware(queryMiddleware);
    
    // 实体获取缓存中间件
    const getMiddleware = {
      name: 'cache.get',
      before: async (ctx: GetContext) => {
        if (!ctx.options.useCache) return;
        
        const cacheKey = this.generateCacheKey('entity', ctx.entityType, ctx.id);
        const cached = await this.cacheProvider!.get(cacheKey);
        
        if (cached) {
          ctx.result = cached;
          ctx.skipExecution = true;
          ctx.fromCache = true;
        }
      },
      after: async (ctx: GetContext) => {
        if (ctx.fromCache || !ctx.options.useCache || !ctx.result) return;
        
        const cacheKey = this.generateCacheKey('entity', ctx.entityType, ctx.id);
        const ttl = ctx.options.cacheTTL || context.moduleConfig.settings.ttl;
        
        await this.cacheProvider!.set(cacheKey, ctx.result, ttl);
      }
    };
    
    context.entityManager.addGetMiddleware(getMiddleware);
  }
  
  private registerCacheInvalidation(context: ModuleContext): void {
    // 实体更新时失效缓存
    const onEntityUpdated = async (event: EntityUpdatedEvent) => {
      const entityCacheKey = this.generateCacheKey('entity', event.entityType, event.entityId);
      await this.cacheProvider!.delete(entityCacheKey);
      
      // 失效相关查询缓存
      const queryPattern = this.generateCacheKey('query', event.entityType, '*');
      await this.cacheProvider!.deletePattern(queryPattern);
    };
    
    // 实体删除时失效缓存
    const onEntityDeleted = async (event: EntityDeletedEvent) => {
      const entityCacheKey = this.generateCacheKey('entity', event.entityType, event.entityId);
      await this.cacheProvider!.delete(entityCacheKey);
      
      // 失效相关查询缓存
      const queryPattern = this.generateCacheKey('query', event.entityType, '*');
      await this.cacheProvider!.deletePattern(queryPattern);
    };
    
    context.eventManager.on('entity.updated', onEntityUpdated);
    context.eventManager.on('entity.deleted', onEntityDeleted);
    
    this.eventListeners.push(
      () => context.eventManager.off('entity.updated', onEntityUpdated),
      () => context.eventManager.off('entity.deleted', onEntityDeleted)
    );
  }
  
  private registerCacheActions(context: ModuleContext): void {
    // 缓存清理动作
    const clearCacheAction = {
      name: 'cache.clear',
      displayName: '清理缓存',
      description: '清理指定类型或所有实体的缓存',
      handler: async (params: { entityType?: string; pattern?: string }) => {
        if (params.pattern) {
          await this.cacheProvider!.deletePattern(params.pattern);
        } else if (params.entityType) {
          const pattern = this.generateCacheKey('*', params.entityType, '*');
          await this.cacheProvider!.deletePattern(pattern);
        } else {
          await this.cacheProvider!.clear();
        }
        
        return { success: true, message: 'Cache cleared successfully' };
      }
    };
    
    // 缓存统计动作
    const cacheStatsAction = {
      name: 'cache.stats',
      displayName: '缓存统计',
      description: '获取缓存使用统计信息',
      handler: async () => {
        return await this.cacheProvider!.getStats();
      }
    };
    
    context.actionManager.register(clearCacheAction);
    context.actionManager.register(cacheStatsAction);
  }
  
  private generateCacheKey(type: string, entityType: string, identifier: string | object): string {
    const prefix = 'ee:cache:';
    
    if (typeof identifier === 'object') {
      identifier = JSON.stringify(identifier);
    }
    
    return `${prefix}${type}:${entityType}:${identifier}`;
  }
}
```

```typescript
export class MyModule implements IEntityModule {
  name = 'my-module';
  version = '1.0.0';
  displayName = 'My Module';
  description = 'Description of my module functionality';
  
  private services: Map<string, any> = new Map();
  private cleanup: Array<() => void> = [];
  
  async initialize(context: ModuleContext): Promise<void> {
    // 初始化服务
    const service1 = new MyService1(context);
    const service2 = new MyService2(context);
    
    await service1.initialize();
    await service2.initialize();
    
    this.services.set('service1', service1);
    this.services.set('service2', service2);
    
    context.logger.info(`${this.displayName} initialized`);
  }
  
  async activate(context: ModuleContext): Promise<void> {
    // 注册服务
    for (const [name, service] of this.services) {
      if (service.activate) {
        await service.activate();
      }
    }
    
    // 注册事件监听器
    this.registerEventListeners(context);
    
    // 注册动作
    this.registerActions(context);
    
    context.logger.info(`${this.displayName} activated`);
  }

    // 清理资源
    this.cleanup.forEach(fn => fn());
    this.cleanup = [];
    
    // 停用服务
    for (const [name, service] of this.services) {
      if (service.deactivate) {
        await service.deactivate();
      }
    }
    
    context.logger.info(`${this.displayName} deactivated`);
  }
  
  async dispose(context: ModuleContext): Promise<void> {
    // 销毁服务
    for (const [name, service] of this.services) {
      if (service.dispose) {
        await service.dispose();
      }
    }
    
    this.services.clear();
    
    context.logger.info(`${this.displayName} disposed`);
  }
```
