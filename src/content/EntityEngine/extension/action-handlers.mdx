# 动作处理器

Entity Engine 的动作处理器系统允许开发者定义和扩展业务逻辑处理。通过创建自定义动作处理器，可以实现复杂的业务流程、数据验证、外部系统集成和自动化工作流。

## 动作处理器架构

### 核心接口

```typescript
/**
 * 动作处理器接口
 */
interface IActionHandler {
  // 基础信息
  name: string;
  displayName: string;
  description?: string;
  version: string;
  category: ActionCategory;
  
  // 支持的实体类型
  supportedEntityTypes?: string[];
  
  // 配置schema
  configSchema: ActionConfigSchema;
  inputSchema: ActionInputSchema;
  outputSchema: ActionOutputSchema;
  
  // 执行方法
  execute(context: ActionExecutionContext): Promise<ActionResult>;
  validate(input: any, config: ActionConfig): Promise<ValidationResult>;
  
  // 生命周期
  initialize?(context: ActionHandlerContext): Promise<void>;
  dispose?(): Promise<void>;
  
  // 权限和安全
  getRequiredPermissions(): string[];
  checkAccess?(context: AccessContext): Promise<boolean>;
  
  // 元数据
  getMetadata(): ActionHandlerMetadata;
}

/**
 * 动作执行上下文
 */
interface ActionExecutionContext {
  // 动作信息
  action: ActionDefinition;
  input: any;
  config: ActionConfig;
  
  // 执行环境
  entityManager: IEntityManager;
  eventManager: IEventManager;
  logger: ILogger;
  
  // 用户上下文
  user: UserContext;
  permissions: PermissionContext;
  session: SessionContext;
  
  // 请求上下文
  requestId: string;
  timestamp: Date;
  metadata: Record<string, any>;
  
  // 工具方法
  transaction: ITransaction;
  cache: ICacheProvider;
  external: IExternalServiceProvider;
}

/**
 * 动作结果
 */
interface ActionResult {
  success: boolean;
  data?: any;
  error?: ActionError;
  metadata?: ActionResultMetadata;
  
  // 副作用
  sideEffects?: SideEffect[];
  events?: ActionEvent[];
  notifications?: Notification[];
}

/**
 * 动作配置schema
 */
interface ActionConfigSchema {
  type: 'object';
  properties: Record<string, SchemaProperty>;
  required?: string[];
  additionalProperties?: boolean;
}

/**
 * 动作分类
 */
enum ActionCategory {
  CREATE = 'create',
  READ = 'read', 
  UPDATE = 'update',
  DELETE = 'delete',
  BUSINESS = 'business',
  INTEGRATION = 'integration',
  WORKFLOW = 'workflow',
  NOTIFICATION = 'notification',
  REPORT = 'report',
  MAINTENANCE = 'maintenance'
}
```

### 动作管理器

```typescript
/**
 * 动作管理器
 */
class ActionManager {
  private handlers = new Map<string, IActionHandler>();
  private categoryHandlers = new Map<ActionCategory, IActionHandler[]>();
  private entityHandlers = new Map<string, IActionHandler[]>();
  private executionQueue: ActionExecution[] = [];
  private middleware: ActionMiddleware[] = [];
  
  register(handler: IActionHandler): void {
    // 验证处理器
    this.validateHandler(handler);
    
    // 检查名称冲突
    if (this.handlers.has(handler.name)) {
      throw new ActionError(`Action handler '${handler.name}' already registered`);
    }
    
    // 注册处理器
    this.handlers.set(handler.name, handler);
    
    // 按类别索引
    if (!this.categoryHandlers.has(handler.category)) {
      this.categoryHandlers.set(handler.category, []);
    }
    this.categoryHandlers.get(handler.category)!.push(handler);
    
    // 按实体类型索引
    if (handler.supportedEntityTypes) {
      for (const entityType of handler.supportedEntityTypes) {
        if (!this.entityHandlers.has(entityType)) {
          this.entityHandlers.set(entityType, []);
        }
        this.entityHandlers.get(entityType)!.push(handler);
      }
    }
    
    // 初始化处理器
    if (handler.initialize) {
      handler.initialize(this.createHandlerContext());
    }
    
    // 触发事件
    this.emitEvent('actionHandlerRegistered', { handler });
  }
  
  unregister(name: string): boolean {
    const handler = this.handlers.get(name);
    if (!handler) return false;
    
    // 检查是否有正在执行的动作
    const runningActions = this.executionQueue.filter(exec => 
      exec.handlerName === name && exec.status === 'running'
    );
    
    if (runningActions.length > 0) {
      throw new ActionError(
        `Cannot unregister handler '${name}': ${runningActions.length} actions still running`
      );
    }
    
    // 销毁处理器
    if (handler.dispose) {
      handler.dispose();
    }
    
    // 从所有索引中移除
    this.handlers.delete(name);
    
    const categoryList = this.categoryHandlers.get(handler.category);
    if (categoryList) {
      const index = categoryList.indexOf(handler);
      if (index > -1) categoryList.splice(index, 1);
    }
    
    if (handler.supportedEntityTypes) {
      for (const entityType of handler.supportedEntityTypes) {
        const entityList = this.entityHandlers.get(entityType);
        if (entityList) {
          const index = entityList.indexOf(handler);
          if (index > -1) entityList.splice(index, 1);
        }
      }
    }
    
    this.emitEvent('actionHandlerUnregistered', { handler });
    return true;
  }
  
  async execute(
    handlerName: string, 
    input: any, 
    config: ActionConfig = {},
    context: Partial<ActionExecutionContext> = {}
  ): Promise<ActionResult> {
    const handler = this.handlers.get(handlerName);
    if (!handler) {
      throw new ActionError(`Action handler '${handlerName}' not found`);
    }
    
    // 创建执行上下文
    const execContext = await this.createExecutionContext(handler, input, config, context);
    
    // 创建执行记录
    const execution: ActionExecution = {
      id: this.generateExecutionId(),
      handlerName,
      input,
      config,
      status: 'pending',
      startTime: new Date(),
      context: execContext
    };
    
    this.executionQueue.push(execution);
    
    try {
      // 权限检查
      await this.checkPermissions(handler, execContext);
      
      // 输入验证
      const validation = await handler.validate(input, config);
      if (!validation.valid) {
        throw new ActionError(`Input validation failed: ${validation.errors.join(', ')}`);
      }
      
      // 执行前中间件
      await this.executeBeforeMiddleware(execContext);
      
      // 更新状态
      execution.status = 'running';
      
      // 执行动作
      const result = await handler.execute(execContext);
      
      // 执行后中间件
      await this.executeAfterMiddleware(execContext, result);
      
      // 更新执行记录
      execution.status = 'completed';
      execution.endTime = new Date();
      execution.result = result;
      
      // 处理副作用
      if (result.sideEffects) {
        await this.processSideEffects(result.sideEffects, execContext);
      }
      
      // 发送事件
      if (result.events) {
        result.events.forEach(event => {
          execContext.eventManager.emit(event.type, event.data);
        });
      }
      
      // 发送通知
      if (result.notifications) {
        await this.processNotifications(result.notifications, execContext);
      }
      
      return result;
      
    } catch (error) {
      // 更新执行记录
      execution.status = 'failed';
      execution.endTime = new Date();
      execution.error = error;
      
      // 执行错误中间件
      await this.executeErrorMiddleware(execContext, error);
      
      // 重新抛出错误
      throw error;
    } finally {
      // 清理执行队列
      this.cleanupExecution(execution.id);
    }
  }
  
  async executeMultiple(
    actions: ActionExecutionRequest[],
    options: MultiActionOptions = {}
  ): Promise<ActionResult[]> {
    const results: ActionResult[] = [];
    
    if (options.parallel) {
      // 并行执行
      const promises = actions.map(action => 
        this.execute(action.handler, action.input, action.config, action.context)
      );
      
      const settled = await Promise.allSettled(promises);
      
      for (const result of settled) {
        if (result.status === 'fulfilled') {
          results.push(result.value);
        } else {
          results.push({
            success: false,
            error: { message: result.reason.message, code: 'EXECUTION_ERROR' }
          });
        }
      }
    } else {
      // 串行执行
      for (const action of actions) {
        try {
          const result = await this.execute(
            action.handler, 
            action.input, 
            action.config, 
            action.context
          );
          results.push(result);
          
          // 如果配置了失败时停止
          if (!result.success && options.stopOnError) {
            break;
          }
        } catch (error) {
          results.push({
            success: false,
            error: { message: error.message, code: 'EXECUTION_ERROR' }
          });
          
          if (options.stopOnError) {
            break;
          }
        }
      }
    }
    
    return results;
  }
  
  get(name: string): IActionHandler | undefined {
    return this.handlers.get(name);
  }
  
  list(): IActionHandler[] {
    return Array.from(this.handlers.values());
  }
  
  getByCategory(category: ActionCategory): IActionHandler[] {
    return this.categoryHandlers.get(category) || [];
  }
  
  getByEntityType(entityType: string): IActionHandler[] {
    return this.entityHandlers.get(entityType) || [];
  }
  
  getExecutionStatus(executionId: string): ActionExecutionStatus | null {
    const execution = this.executionQueue.find(exec => exec.id === executionId);
    return execution ? {
      id: execution.id,
      status: execution.status,
      progress: execution.progress,
      startTime: execution.startTime,
      endTime: execution.endTime,
      error: execution.error
    } : null;
  }
  
  addMiddleware(middleware: ActionMiddleware): void {
    this.middleware.push(middleware);
    this.middleware.sort((a, b) => (b.priority || 0) - (a.priority || 0));
  }
  
  removeMiddleware(name: string): boolean {
    const index = this.middleware.findIndex(m => m.name === name);
    if (index > -1) {
      this.middleware.splice(index, 1);
      return true;
    }
    return false;
  }
  
  // 私有方法
  private validateHandler(handler: IActionHandler): void {
    const errors: string[] = [];
    
    if (!handler.name) {
      errors.push('Handler must have a name');
    }
    
    if (!handler.displayName) {
      errors.push('Handler must have a display name');
    }
    
    if (!handler.version) {
      errors.push('Handler must have a version');
    }
    
    if (!handler.category) {
      errors.push('Handler must have a category');
    }
    
    if (typeof handler.execute !== 'function') {
      errors.push('Handler must implement execute method');
    }
    
    if (typeof handler.validate !== 'function') {
      errors.push('Handler must implement validate method');
    }
    
    if (errors.length > 0) {
      throw new ActionError(`Invalid action handler: ${errors.join(', ')}`);
    }
  }
  
  private async createExecutionContext(
    handler: IActionHandler,
    input: any,
    config: ActionConfig,
    context: Partial<ActionExecutionContext>
  ): Promise<ActionExecutionContext> {
    return {
      action: {
        name: handler.name,
        displayName: handler.displayName,
        category: handler.category
      },
      input,
      config,
      entityManager: context.entityManager || this.getEntityManager(),
      eventManager: context.eventManager || this.getEventManager(),
      logger: context.logger || this.getLogger(),
      user: context.user || this.getCurrentUser(),
      permissions: context.permissions || this.getCurrentPermissions(),
      session: context.session || this.getCurrentSession(),
      requestId: context.requestId || this.generateRequestId(),
      timestamp: new Date(),
      metadata: context.metadata || {},
      transaction: context.transaction || this.createTransaction(),
      cache: context.cache || this.getCacheProvider(),
      external: context.external || this.getExternalServiceProvider()
    };
  }
  
  private async checkPermissions(handler: IActionHandler, context: ActionExecutionContext): Promise<void> {
    const requiredPermissions = handler.getRequiredPermissions();
    
    for (const permission of requiredPermissions) {
      if (!context.permissions.has(permission)) {
        throw new ActionError(`Missing required permission: ${permission}`);
      }
    }
    
    if (handler.checkAccess) {
      const hasAccess = await handler.checkAccess({
        user: context.user,
        permissions: context.permissions,
        action: handler.name,
        input: context.input
      });
      
      if (!hasAccess) {
        throw new ActionError('Access denied');
      }
    }
  }
  
  private async executeBeforeMiddleware(context: ActionExecutionContext): Promise<void> {
    for (const middleware of this.middleware) {
      if (middleware.before) {
        await middleware.before(context);
      }
    }
  }
  
  private async executeAfterMiddleware(context: ActionExecutionContext, result: ActionResult): Promise<void> {
    for (const middleware of this.middleware) {
      if (middleware.after) {
        await middleware.after(context, result);
      }
    }
  }
  
  private async executeErrorMiddleware(context: ActionExecutionContext, error: Error): Promise<void> {
    for (const middleware of this.middleware) {
      if (middleware.onError) {
        await middleware.onError(context, error);
      }
    }
  }
  
  private async processSideEffects(sideEffects: SideEffect[], context: ActionExecutionContext): Promise<void> {
    for (const sideEffect of sideEffects) {
      try {
        await this.executeSideEffect(sideEffect, context);
      } catch (error) {
        context.logger.error(`Side effect execution failed: ${error.message}`);
        // 副作用失败不影响主要操作
      }
    }
  }
  
  private async processNotifications(notifications: Notification[], context: ActionExecutionContext): Promise<void> {
    for (const notification of notifications) {
      try {
        await this.sendNotification(notification, context);
      } catch (error) {
        context.logger.error(`Notification sending failed: ${error.message}`);
      }
    }
  }
  
  private generateExecutionId(): string {
    return 'exec_' + Math.random().toString(36).substr(2, 12);
  }
  
  private generateRequestId(): string {
    return 'req_' + Math.random().toString(36).substr(2, 12);
  }
  
  private cleanupExecution(executionId: string): void {
    const index = this.executionQueue.findIndex(exec => exec.id === executionId);
    if (index > -1) {
      this.executionQueue.splice(index, 1);
    }
  }
  
  private createHandlerContext(): ActionHandlerContext {
    return {
      actionManager: this,
      entityManager: this.getEntityManager(),
      eventManager: this.getEventManager(),
      logger: this.getLogger()
    };
  }
  
  private emitEvent(eventName: string, data: any): void {
    // 触发事件
  }
  
  // 辅助方法
  private getEntityManager(): IEntityManager { 
    // 实现
    return {} as IEntityManager; 
  }
  
  private getEventManager(): IEventManager { 
    // 实现
    return {} as IEventManager; 
  }
  
  private getLogger(): ILogger { 
    // 实现
    return {} as ILogger; 
  }
  
  private getCurrentUser(): UserContext { 
    // 实现
    return {} as UserContext; 
  }
  
  private getCurrentPermissions(): PermissionContext { 
    // 实现
    return {} as PermissionContext; 
  }
  
  private getCurrentSession(): SessionContext { 
    // 实现
    return {} as SessionContext; 
  }
  
  private createTransaction(): ITransaction { 
    // 实现
    return {} as ITransaction; 
  }
  
  private getCacheProvider(): ICacheProvider { 
    // 实现
    return {} as ICacheProvider; 
  }
  
  private getExternalServiceProvider(): IExternalServiceProvider { 
    // 实现
    return {} as IExternalServiceProvider; 
  }
  
  private async executeSideEffect(sideEffect: SideEffect, context: ActionExecutionContext): Promise<void> {
    // 实现副作用执行
  }
  
  private async sendNotification(notification: Notification, context: ActionExecutionContext): Promise<void> {
    // 实现通知发送
  }
}
```

## 自定义动作处理器示例

### 邮件发送动作处理器

```typescript
/**
 * 邮件发送动作处理器
 */
class EmailActionHandler implements IActionHandler {
  name = 'send-email';
  displayName = '发送邮件';
  description = '发送电子邮件通知';
  version = '1.0.0';
  category = ActionCategory.NOTIFICATION;
  
  supportedEntityTypes = ['*'];
  
  configSchema: ActionConfigSchema = {
    type: 'object',
    properties: {
      smtpHost: {
        type: 'string',
        title: 'SMTP主机',
        required: true
      },
      smtpPort: {
        type: 'number',
        title: 'SMTP端口',
        default: 587
      },
      username: {
        type: 'string',
        title: '用户名',
        required: true
      },
      password: {
        type: 'string',
        title: '密码',
        format: 'password',
        required: true
      },
      fromAddress: {
        type: 'string',
        title: '发件人地址',
        format: 'email',
        required: true
      },
      fromName: {
        type: 'string',
        title: '发件人姓名'
      },
      enableTLS: {
        type: 'boolean',
        title: '启用TLS',
        default: true
      }
    }
  };
  
  inputSchema: ActionInputSchema = {
    type: 'object',
    properties: {
      to: {
        oneOf: [
          { type: 'string', format: 'email' },
          { 
            type: 'array', 
            items: { type: 'string', format: 'email' },
            minItems: 1
          }
        ],
        title: '收件人',
        required: true
      },
      cc: {
        type: 'array',
        items: { type: 'string', format: 'email' },
        title: '抄送'
      },
      bcc: {
        type: 'array', 
        items: { type: 'string', format: 'email' },
        title: '密送'
      },
      subject: {
        type: 'string',
        title: '主题',
        required: true,
        maxLength: 998
      },
      htmlBody: {
        type: 'string',
        title: 'HTML内容'
      },
      textBody: {
        type: 'string',
        title: '纯文本内容'
      },
      attachments: {
        type: 'array',
        items: {
          type: 'object',
          properties: {
            filename: { type: 'string' },
            contentType: { type: 'string' },
            content: { type: 'string' }, // Base64 encoded
            path: { type: 'string' }     // File path
          },
          required: ['filename'],
          oneOf: [
            { required: ['content'] },
            { required: ['path'] }
          ]
        },
        title: '附件'
      },
      priority: {
        type: 'string',
        enum: ['high', 'normal', 'low'],
        title: '优先级',
        default: 'normal'
      },
      template: {
        type: 'object',
        properties: {
          name: { type: 'string' },
          variables: { type: 'object' }
        },
        title: '邮件模板'
      }
    },
    required: ['to', 'subject']
  };
  
  outputSchema: ActionOutputSchema = {
    type: 'object',
    properties: {
      messageId: { type: 'string' },
      accepted: { 
        type: 'array', 
        items: { type: 'string', format: 'email' } 
      },
      rejected: { 
        type: 'array', 
        items: { type: 'string', format: 'email' } 
      },
      response: { type: 'string' },
      envelope: {
        type: 'object',
        properties: {
          from: { type: 'string' },
          to: { type: 'array', items: { type: 'string' } }
        }
      }
    }
  };
  
  private transporter?: nodemailer.Transporter;
  private templateEngine?: ITemplateEngine;
  
  async initialize(context: ActionHandlerContext): Promise<void> {
    this.templateEngine = new HandlebarsTemplateEngine();
    await this.templateEngine.initialize();
    
    context.logger.info('Email action handler initialized');
  }
  
  async dispose(): Promise<void> {
    if (this.transporter) {
      this.transporter.close();
      this.transporter = undefined;
    }
    
    if (this.templateEngine) {
      await this.templateEngine.dispose();
      this.templateEngine = undefined;
    }
  }
  
  async execute(context: ActionExecutionContext): Promise<ActionResult> {
    const { input, config, logger } = context;
    
    try {
      // 初始化邮件传输器
      await this.initializeTransporter(config);
      
      // 准备邮件内容
      const emailContent = await this.prepareEmailContent(input, context);
      
      // 发送邮件
      const result = await this.sendEmail(emailContent);
      
      // 记录发送结果
      logger.info(`Email sent successfully: ${result.messageId}`);
      
      // 创建审计日志
      const auditEvent = {
        type: 'email.sent',
        data: {
          to: Array.isArray(input.to) ? input.to : [input.to],
          subject: input.subject,
          messageId: result.messageId,
          timestamp: new Date()
        }
      };
      
      return {
        success: true,
        data: {
          messageId: result.messageId,
          accepted: result.accepted,
          rejected: result.rejected,
          response: result.response,
          envelope: result.envelope
        },
        metadata: {
          sentAt: new Date(),
          provider: 'smtp'
        },
        events: [auditEvent]
      };
      
    } catch (error) {
      logger.error(`Email sending failed: ${error.message}`);
      
      return {
        success: false,
        error: {
          message: `Failed to send email: ${error.message}`,
          code: 'EMAIL_SEND_ERROR',
          details: error
        }
      };
    }
  }
  
  async validate(input: any, config: ActionConfig): Promise<ValidationResult> {
    const errors: string[] = [];
    const warnings: string[] = [];
    
    // 基础验证
    if (!input.to) {
      errors.push('Recipient address is required');
    }
    
    if (!input.subject || input.subject.trim() === '') {
      errors.push('Subject is required');
    }
    
    if (!input.htmlBody && !input.textBody) {
      errors.push('Either HTML body or text body is required');
    }
    
    // 邮件地址验证
    const emailAddresses = this.extractEmailAddresses(input);
    for (const email of emailAddresses) {
      if (!this.isValidEmail(email)) {
        errors.push(`Invalid email address: ${email}`);
      }
    }
    
    // 附件验证
    if (input.attachments) {
      for (const attachment of input.attachments) {
        if (!attachment.filename) {
          errors.push('Attachment filename is required');
        }
        
        if (!attachment.content && !attachment.path) {
          errors.push('Attachment must have either content or path');
        }
        
        if (attachment.content && attachment.path) {
          warnings.push('Attachment has both content and path, content will be used');
        }
      }
    }
    
    // 配置验证
    if (!config.smtpHost) {
      errors.push('SMTP host is required in configuration');
    }
    
    if (!config.username) {
      errors.push('SMTP username is required in configuration');
    }
    
    if (!config.password) {
      errors.push('SMTP password is required in configuration');
    }
    
    if (!config.fromAddress || !this.isValidEmail(config.fromAddress)) {
      errors.push('Valid sender address is required in configuration');
    }
    
    return { valid: errors.length === 0, errors, warnings };
  }
  
  getRequiredPermissions(): string[] {
    return ['send_email'];
  }
  
  async checkAccess(context: AccessContext): Promise<boolean> {
    // 检查用户是否有发送邮件的权限
    if (!context.permissions.has('send_email')) {
      return false;
    }
    
    // 检查发送频率限制
    const rateLimitKey = `email_rate_limit:${context.user.id}`;
    const recentSends = await this.getRateLimitCount(rateLimitKey);
    
    if (recentSends > 100) { // 每小时最多100封邮件
      return false;
    }
    
    return true;
  }
  
  getMetadata(): ActionHandlerMetadata {
    return {
      async: true,
      timeout: 30000,
      retryable: true,
      maxRetries: 3,
      rateLimited: true,
      rateLimit: {
        requests: 100,
        windowMs: 3600000 // 1 hour
      }
    };
  }
  
  private async initializeTransporter(config: ActionConfig): Promise<void> {
    if (this.transporter) return;
    
    const transportConfig = {
      host: config.smtpHost,
      port: config.smtpPort || 587,
      secure: config.smtpPort === 465,
      auth: {
        user: config.username,
        pass: config.password
      },
      tls: {
        rejectUnauthorized: false
      }
    };
    
    this.transporter = nodemailer.createTransporter(transportConfig);
    
    // 验证连接
    await this.transporter.verify();
  }
  
  private async prepareEmailContent(input: any, context: ActionExecutionContext): Promise<EmailContent> {
    let htmlBody = input.htmlBody;
    let textBody = input.textBody;
    
    // 处理模板
    if (input.template && this.templateEngine) {
      const template = await this.loadTemplate(input.template.name);
      const templateData = {
        ...input.template.variables,
        user: context.user,
        timestamp: new Date().toISOString()
      };
      
      htmlBody = await this.templateEngine.render(template.html, templateData);
      textBody = await this.templateEngine.render(template.text, templateData);
    }
    
    // 处理附件
    const attachments = [];
    if (input.attachments) {
      for (const att of input.attachments) {
        if (att.content) {
          attachments.push({
            filename: att.filename,
            content: Buffer.from(att.content, 'base64'),
            contentType: att.contentType
          });
        } else if (att.path) {
          attachments.push({
            filename: att.filename,
            path: att.path,
            contentType: att.contentType
          });
        }
      }
    }
    
    return {
      from: {
        address: context.config.fromAddress,
        name: context.config.fromName || context.config.fromAddress
      },
      to: Array.isArray(input.to) ? input.to : [input.to],
      cc: input.cc || [],
      bcc: input.bcc || [],
      subject: input.subject,
      html: htmlBody,
      text: textBody,
      attachments,
      priority: input.priority || 'normal',
      headers: {
        'X-Action-Handler': this.name,
        'X-Request-ID': context.requestId
      }
    };
  }
  
  private async sendEmail(content: EmailContent): Promise<any> {
    if (!this.transporter) {
      throw new Error('Email transporter not initialized');
    }
    
    const mailOptions = {
      from: `"${content.from.name}" <${content.from.address}>`,
      to: content.to.join(', '),
      cc: content.cc.length > 0 ? content.cc.join(', ') : undefined,
      bcc: content.bcc.length > 0 ? content.bcc.join(', ') : undefined,
      subject: content.subject,
      html: content.html,
      text: content.text,
      attachments: content.attachments,
      priority: content.priority,
      headers: content.headers
    };
    
    return await this.transporter.sendMail(mailOptions);
  }
  
  private extractEmailAddresses(input: any): string[] {
    const addresses: string[] = [];
    
    if (input.to) {
      if (Array.isArray(input.to)) {
        addresses.push(...input.to);
      } else {
        addresses.push(input.to);
      }
    }
    
    if (input.cc) {
      addresses.push(...input.cc);
    }
    
    if (input.bcc) {
      addresses.push(...input.bcc);
    }
    
    return addresses;
  }
  
  private isValidEmail(email: string): boolean {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return emailRegex.test(email);
  }
  
  private async loadTemplate(templateName: string): Promise<EmailTemplate> {
    // 从数据库或文件系统加载模板
    // 这里是简化实现
    return {
      name: templateName,
      html: '<div>{{content}}</div>',
      text: '{{content}}'
    };
  }
  
  private async getRateLimitCount(key: string): Promise<number> {
    // 从缓存获取频率限制计数
    // 这里是简化实现
    return 0;
  }
}
```

```typescript
/**
 * 数据导出动作处理器

class DataExportActionHandler implements IActionHandler {
  name = 'export-data';
  displayName = '数据导出';
  description = '将实体数据导出为指定格式';
  version = '1.0.0';
  category = ActionCategory.REPORT;
  
  supportedEntityTypes = ['*'];
  
  configSchema: ActionConfigSchema = {
    type: 'object',
    properties: {
      maxRecords: {
        type: 'number',
        title: '最大记录数',
        minimum: 1,
        maximum: 100000,
        default: 10000
      },
      timeout: {
        type: 'number',
        title: '超时时间（秒）',
        minimum: 30,
        maximum: 3600,
        default: 300
      },
      storageProvider: {
        type: 'string',
        enum: ['local', 's3', 'gcs', 'azure'],
        title: '存储提供商',
        default: 'local'
      },
      storageConfig: {
        type: 'object',
        title: '存储配置'
      }
    }
  };
  
  inputSchema: ActionInputSchema = {
    type: 'object',
    properties: {
      entityType: {
        type: 'string',
        title: '实体类型',
        required: true
      },
      format: {
        type: 'string',
        enum: ['csv', 'xlsx', 'json', 'xml', 'pdf'],
        title: '导出格式',
        default: 'csv'
      },
      query: {
        type: 'object',
        properties: {
          filter: { type: 'object' },
          sort: { type: 'array' },
          fields: { 
            type: 'array', 
            items: { type: 'string' },
            title: '导出字段'
          }
        },
        title: '查询条件'
      },
      options: {
        type: 'object',
        properties: {
          filename: { 
            type: 'string',
            title: '文件名（不含扩展名）'
          },
          includeHeaders: {
            type: 'boolean',
            title: '包含表头',
            default: true
          },
          dateFormat: {
            type: 'string',
            title: '日期格式',
            default: 'YYYY-MM-DD'
          },
          encoding: {
            type: 'string',
            enum: ['utf8', 'gbk'],
            title: '文件编码',
            default: 'utf8'
          }
        },
        title: '导出选项'
      }
    },
    required: ['entityType']
  };
  
  outputSchema: ActionOutputSchema = {
    type: 'object',
    properties: {
      fileUrl: { type: 'string' },
      fileName: { type: 'string' },
      fileSize: { type: 'number' },
      recordCount: { type: 'number' },
      format: { type: 'string' },
      downloadToken: { type: 'string' },
      expiresAt: { type: 'string', format: 'date-time' }
    }
  };
  
  private exportProcessors = new Map<string, IExportProcessor>();
  private storageProviders = new Map<string, IStorageProvider>();
  
  async initialize(context: ActionHandlerContext): Promise<void> {
    // 注册导出处理器
    this.exportProcessors.set('csv', new CSVExportProcessor());
    this.exportProcessors.set('xlsx', new ExcelExportProcessor());
    this.exportProcessors.set('json', new JSONExportProcessor());
    this.exportProcessors.set('xml', new XMLExportProcessor());
    this.exportProcessors.set('pdf', new PDFExportProcessor());
    
    // 注册存储提供商
    this.storageProviders.set('local', new LocalStorageProvider());
    this.storageProviders.set('s3', new S3StorageProvider());
    this.storageProviders.set('gcs', new GCSStorageProvider());
    this.storageProviders.set('azure', new AzureStorageProvider());
    
    // 初始化所有处理器和提供商
    for (const processor of this.exportProcessors.values()) {
      await processor.initialize();
    }
    
    for (const provider of this.storageProviders.values()) {
      await provider.initialize();
    }
    
    context.logger.info('Data export action handler initialized');
  }
  
  async dispose(): Promise<void> {
    // 清理资源
    for (const processor of this.exportProcessors.values()) {
      await processor.dispose();
    }
    
    for (const provider of this.storageProviders.values()) {
      await provider.dispose();
    }
    
    this.exportProcessors.clear();
    this.storageProviders.clear();
  }
  
  async execute(context: ActionExecutionContext): Promise<ActionResult> {
    const { input, config, logger, entityManager } = context;
    
    try {
      // 获取导出处理器
      const processor = this.exportProcessors.get(input.format || 'csv');
      if (!processor) {
        throw new Error(`Unsupported export format: ${input.format}`);
      }
      
      // 获取存储提供商
      const storageProvider = this.storageProviders.get(config.storageProvider || 'local');
      if (!storageProvider) {
        throw new Error(`Unsupported storage provider: ${config.storageProvider}`);
      }
      
      // 查询数据
      const queryResult = await this.queryData(input, config, entityManager);
      
      // 处理导出
      const exportResult = await this.processExport(queryResult, input, processor, context);
      
      // 上传文件
      const uploadResult = await this.uploadFile(exportResult, storageProvider, config);
      
      // 生成下载令牌
      const downloadToken = this.generateDownloadToken(uploadResult.fileUrl, context.user.id);
      
      logger.info(`Data export completed: ${uploadResult.fileName} (${queryResult.totalRecords} records)`);
      
      return {
        success: true,
        data: {
          fileUrl: uploadResult.fileUrl,
          fileName: uploadResult.fileName,
          fileSize: uploadResult.fileSize,
          recordCount: queryResult.totalRecords,
          format: input.format || 'csv',
          downloadToken,
          expiresAt: new Date(Date.now() + 24 * 60 * 60 * 1000) // 24小时后过期
        },
        metadata: {
          exportedAt: new Date(),
          processor: processor.name,
          storageProvider: storageProvider.name
        },
        events: [{
          type: 'data.exported',
          data: {
            entityType: input.entityType,
            format: input.format,
            recordCount: queryResult.totalRecords,
            userId: context.user.id,
            timestamp: new Date()
          }
        }]
      };
      
    } catch (error) {
      logger.error(`Data export failed: ${error.message}`);
      
      return {
        success: false,
        error: {
          message: `Export failed: ${error.message}`,
          code: 'EXPORT_ERROR',
          details: error
        }
      };
    }
  }
  
  async validate(input: any, config: ActionConfig): Promise<ValidationResult> {
    const errors: string[] = [];
    const warnings: string[] = [];
    
    if (!input.entityType) {
      errors.push('Entity type is required');
    }
    
    if (input.format && !this.exportProcessors.has(input.format)) {
      errors.push(`Unsupported export format: ${input.format}`);
    }
    
    if (config.storageProvider && !this.storageProviders.has(config.storageProvider)) {
      errors.push(`Unsupported storage provider: ${config.storageProvider}`);
    }
    
    if (input.query?.fields && !Array.isArray(input.query.fields)) {
      errors.push('Query fields must be an array');
    }
    
    if (config.maxRecords && input.query?.limit && input.query.limit > config.maxRecords) {
      warnings.push(`Query limit (${input.query.limit}) exceeds max records (${config.maxRecords})`);
    }
    
    return { valid: errors.length === 0, errors, warnings };
  }
  
  getRequiredPermissions(): string[] {
    return ['export_data'];
  }
  
  async checkAccess(context: AccessContext): Promise<boolean> {
    // 检查实体类型导出权限
    const entityType = context.input.entityType;
    if (!context.permissions.has(`export_data:${entityType}`)) {
      return false;
    }
    
    // 检查导出频率限制
    const rateLimitKey = `export_rate_limit:${context.user.id}`;
    const recentExports = await this.getRateLimitCount(rateLimitKey);
    
    if (recentExports > 10) { // 每小时最多10次导出
      return false;
    }
    
    return true;
  }
  
  getMetadata(): ActionHandlerMetadata {
    return {
      async: true,
      timeout: 300000, // 5分钟
      retryable: false,
      rateLimited: true,
      rateLimit: {
        requests: 10,
        windowMs: 3600000 // 1小时
      }
    };
  }
  
  private async queryData(
    input: any, 
    config: ActionConfig, 
    entityManager: IEntityManager
  ): Promise<QueryResult> {
    const query = {
      entityType: input.entityType,
      filter: input.query?.filter || {},
      sort: input.query?.sort || [],
      fields: input.query?.fields,
      limit: Math.min(input.query?.limit || config.maxRecords, config.maxRecords),
      offset: input.query?.offset || 0
    };
    
    const result = await entityManager.query(query);
    
    return {
      records: result.data,
      totalRecords: result.total,
      fields: input.query?.fields || this.inferFields(result.data)
    };
  }
  
  private async processExport(
    queryResult: QueryResult,
    input: any,
    processor: IExportProcessor,
    context: ActionExecutionContext
  ): Promise<ExportResult> {
    const options: ExportOptions = {
      filename: input.options?.filename || `${input.entityType}_export_${new Date().getTime()}`,
      includeHeaders: input.options?.includeHeaders !== false,
      dateFormat: input.options?.dateFormat || 'YYYY-MM-DD',
      encoding: input.options?.encoding || 'utf8',
      metadata: {
        entityType: input.entityType,
        exportedBy: context.user.name || context.user.id,
        exportedAt: new Date(),
        totalRecords: queryResult.totalRecords
      }
    };
    
    const buffer = await processor.process(queryResult.records, queryResult.fields, options);
    
    return {
      buffer,
      filename: `${options.filename}.${processor.getFileExtension()}`,
      contentType: processor.getContentType(),
      size: buffer.length
    };
  }
  
  private async uploadFile(
    exportResult: ExportResult,
    storageProvider: IStorageProvider,
    config: ActionConfig
  ): Promise<UploadResult> {
    const uploadOptions = {
      filename: exportResult.filename,
      contentType: exportResult.contentType,
      metadata: {
        size: exportResult.size,
        uploadedAt: new Date()
      },
      ...config.storageConfig
    };
    
    return await storageProvider.upload(exportResult.buffer, uploadOptions);
  }
  
  private inferFields(records: any[]): string[] {
    if (records.length === 0) return [];
    
    const fieldSet = new Set<string>();
    records.slice(0, 5).forEach(record => {
      Object.keys(record).forEach(key => fieldSet.add(key));
    });
    
    return Array.from(fieldSet);
  }
  
  private generateDownloadToken(fileUrl: string, userId: string): string {
    // 生成安全的下载令牌
    const payload = {
      fileUrl,
      userId,
      expiresAt: Date.now() + 24 * 60 * 60 * 1000 // 24小时
    };
    
    return Buffer.from(JSON.stringify(payload)).toString('base64');
  }
  
  private async getRateLimitCount(key: string): Promise<number> {
    // 获取频率限制计数
    return 0;
  }
}
```

```typescript
/**

const auditMiddleware: ActionMiddleware = {
  name: 'audit',
  priority: 100,
  
  async before(context: ActionExecutionContext) {
    // 记录动作开始
    context.logger.info(`Action started: ${context.action.name}`, {
      user: context.user.id,
      input: context.input,
      timestamp: context.timestamp
    });
  },
  
  async after(context: ActionExecutionContext, result: ActionResult) {
    // 记录动作结果

      duration: Date.now() - context.timestamp.getTime()

    await context.entityManager.create('ActionAudit', {
      actionName: context.action.name,

      input: JSON.stringify(context.input),
      result: JSON.stringify(result),
      success: result.success,
      executedAt: context.timestamp
    });
  },
  
  async onError(context: ActionExecutionContext, error: Error) {
    // 记录错误
    context.logger.error(`Action failed: ${context.action.name}`, {
      error: error.message,
      stack: error.stack,
      user: context.user.id

  }
```
